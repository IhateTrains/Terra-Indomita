    title = me_colchis_21_title
    desc = me_colchis_21_desc  
    picture = looting
    left_portrait = current_ruler
    trigger = {
        always = yes
    }

    option = { # Keep treasure in reliquary
        name = me_colchis_21.a
        create_country_treasure = {
            key = golden_goat_rhyton
            icon = treasure_chest
            modifier = {
                global_nobles_desired_pop_ratio = 0.01
	            global_pop_promotion_speed_modifier = 0.05
            }
        }
    }

    option = { # The king's toy
        name = me_colchis_21.b
        current_ruler = {
            add_popularity = 5
            add_gold = 100
            add_corruption = 5
            add_charisma = 2
        }
        add_legitimacy = 5
    }

    option = { # Sell treasure
        name = me_colchis_21.c
        add_treasury = 200
        add_political_influence = 50
    }
}
