me_jomon.13 = { # Gods of the North
    type = country_event

    title = me_jomon_13_title
    desc = me_jomon_13_desc
    picture = celtic_religious_festival

    option = {
        name = "me_jomon_13.a"
        add_country_modifier = {
            name = nivkh_syncretism
            duration = 3650
        }
        custom_tooltip = me_jomon_13_tooltip
    }
    option = {
        name = "me_jomon_13.b"
        set_country_religion = nivkh_religion
        change_religion_effect = {
            RELIGION = nivkh_religion
        }
        while = {
            count = 10
            random_owned_province = {
                random_pops_in_province = {
                    limit = {
                        NOT = { pop_religion = nivkh_religion }
                    }
                    set_pop_religion = nivkh_religion
                }
            }
        }
        custom_tooltip = me_jomon_13_tooltip
    }
    option = {
        name = "me_jomon_13.c"
        c:TU7 = {
            add_country_modifier = {
                name = recent_convert_modifier
                duration = 3650
            }
            set_country_religion = ainu_religion
        }
        custom_tooltip = me_jomon_13_tooltip
    }
}
