        custom_tooltip = me_ezo_13_tooltip
    }
    option = {
        name = "me_ezo_13.b"
        set_country_religion = nivkh_religion
        change_religion_effect = {
            RELIGION = nivkh_religion
        }
        while = {
            count = 10
            random_owned_province = {
                random_pops_in_province = {
                    limit = {
                        NOT = { pop_religion = nivkh_religion }

                    set_pop_religion = nivkh_religion
                }
            }
        }
        custom_tooltip = me_ezo_13_tooltip
    }
    option = {
        name = "me_ezo_13.c"

            add_country_modifier = {
                name = recent_convert_modifier
                duration = 3650
            }
