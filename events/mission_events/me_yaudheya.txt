me_yaudheya.1 = { # Every Kshatriya a King
    type = country_event

    title = me_yaudheya_1_title
    desc = me_yaudheya_1_desc
    picture = greek_court

    option = {
        name = "me_yaudheya_1.a"
        custom_tooltip = me_yaudheya_1_tooltip
        oligarchic_approval_effect = {
            VALUE = party_approval_large
        }
        if = {
            limit = {
                current_ruler = {
                    martial < 2
                }
            }
            current_ruler = {
                add_martial = 9
            }
        }
        if = {
            limit = {
                AND = {
                    current_ruler = {
                        martial >= 2
                    }
                    current_ruler = {
                        martial < 7
                    }
                }
            }
            current_ruler = {
                add_martial = 5
            }
        }
        if = {
            limit = {
                current_ruler = {
                    martial >= 7
                }
            }
            current_ruler = {
                add_martial = 2
            }
        }
        add_stability = 5
        add_political_influence = 25
    }
}

        name = "me_yaudheya_70.a"
        current_ruler = {
            remove_all_positions = yes
            remove_all_offices = yes
            move_country = c:overlord
            save_scope_as = yaudheya_warrior
        }
        if = {
            limit = {
                overlord = {
                    is_ai = no
                }
            }
            overlord = {
                trigger_event = {
                    id = me_yaudheya.72
                    days = { 5 10 }
                }
            }
        }
    }
    option = {
        name = "me_yaudheya_70.b"
        reverse_add_opinion = {
            target = overlord
            modifier = opinion_insult
        }
        if = {
            limit = {
                overlord = {

                }
            }
            overlord = {
                trigger_event = {
                    id = me_yaudheya.71
                    days = { 5 10 }
                }
            }
        }
    }
