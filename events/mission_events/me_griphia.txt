			}
		}
    }

    type = country_event

    desc = me_griphia_15_desc
    picture = great_library
	left_portrait = root.current_ruler
	right_portrait = scope:delos_owner.current_ruler

	immediate = {
		if = {
			limit = {
				p:1835 = {
					has_owner = yes
				}
			}
			p:1835 = {
				owner = {
					save_scope_as = delos_owner
				}
			}
		}
	}

    option = {
        name = "me_griphia_15.a"
		add_innovation = 2
		hidden_effect = {
			every_country = {
				limit = {
					country_culture_group = hellenic

						this = root
					}
				}
				add_opinion = {
					modifier = gryphian_envoys
					target = root
				}
			}
		}
    }
}
