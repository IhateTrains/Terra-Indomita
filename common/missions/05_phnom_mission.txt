    phnom_mission_tree_task_2 = { #Mauryan Marriage
        icon = "task_battle"
		requires = { phnom_mission_tree_task_1 }
		duration = 90
		
		highlight = {
            scope:province = {
                province_id = 8773
            }
        }
		
        allow = {
			trigger_if = {
                limit = {
                    c:MRY = { has_land = yes }
                }
                trigger_if = {
                    limit = {
                        char:66 = { is_alive = yes}
                    }
                    char:66 = { age <= 75 }
                }
                trigger_else_if = {
                    limit = {
                        char:66 = { is_alive = no }
                    }
                    current_ruler = {
                        any_child = {
                            is_alive = yes
                            is_male = yes
                        }
                    }
                }
            	c:MRY = {
            		opinion = {
            			target = c:PHN
            			value >= 30
            		}
                    any_character = {
                        has_trait = maurya
                        is_female = yes
                        age <= 40
                        age > 15
                    }
            	}
            }
            trigger_else = {
                c:MRY = { has_land = no }
            }
        }
		
        on_completion = {
            if = {
                limit = {
                    c:MRY = { has_land = yes }
                }
                trigger_event = me_phnom.4
                if = {
                    limit = {
                        c:MRY = { has_land = yes }
                        char:66 = { is_alive = yes }
                    }
                    #Marry an Achaemenid character to Zipoetes son if he is alive, otherwise marry a random child.
                    custom_tooltip  = phnom_missions_task_2_tt
                    if = {
                        limit = {
                            current_ruler = { is_married = yes }
                        }
                        current_ruler = { divorce_character = spouse }
                    }
                    c:MRY = {
                        random_character = {
                            limit = {
                                has_trait = maurya
                                is_female = yes
                                age <= 40
                                age > 15
                            }
                            if = {
                                limit = { is_married = yes }
                                divorce_character = spouse
                            }
                            save_scope_as = marriage_partner
                        }
                    }
                    current_ruler = { marry_character = scope:marriage_partner }
                }
                else_if = {
                    limit = {
                        c:MRY = { has_land = yes }
                        current_ruler = {
                            any_child = {
                                is_alive = yes
                                is_male = yes
                                age >= 16
                            }
                        }
                    }
                    c:MRY = {
                        random_character = {
                            limit = {
                                has_trait = maurya
                                is_female = yes
                                age <= 40
                                age > 15
                            }
                            if = {
                                limit = { is_married = yes }
                                divorce_character = spouse
                            }
                            save_scope_as = marriage_partner
                        }
                    }
                    current_ruler = {
                        ordered_child = {
                            limit = {
                                age >= 16
                                is_male = yes
                                is_alive = yes
                            }
                            order_by = age
                            position = end
                            if = {
                                limit = { is_married = yes }
                                divorce_character = spouse
                            }
                            marry_character = scope:marriage_partner
                        }
                    }
                }
            }
            else = {
                add_treasury = 100
                current_ruler = {
                    add_popularity = 5
                }
            }
			add_country_modifier = {
				name = phnom_maurya_marriage_trade_connection_mod
				duration = -1
			}
        }
    }
